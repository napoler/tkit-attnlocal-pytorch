[文档]    def autoBulidlMaskLimit(self):
        """
        构建一个矩阵
        自动限制 预测长度


        """
        datas = []
        for it in range(self.maxlen):
            new = it * [0] + [1] * self.limit + [0] * self.maxlen
            datas.append(new[:self.maxlen + self.limit])
        return datas
